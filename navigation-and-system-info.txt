# pwd

Print the current working directory.

pwd
# cd

Change the current directory.

cd /var/log
# ls

List files and directories.

ls -la
# history

Display previously executed commands.

history
# clear

Clear the terminal screen.

clear
# date

Display the current system date and time.

date
# uptime

Show system uptime and load averages.

uptime
# uname

Display kernel and operating system information.

uname -a
# cat /etc/os-release

Display Linux distribution information.

cat /etc/os-release
